Спроектируйте структуру базы данных, в которой будут храниться товары, клиенты и заказы.
Результат ожидается в виде DDL команды (SQL запрос на создание таблицы) для MySQL.

CREATE DATABASE eshop;

CREATE TABLE customers (
    `id` int NOT NULL AUTO_INCREMENT,
    `login` varchar(255) NOT NULL,
    `password` varchar(255) NOT NULL,
    `name` varchar(255) NOT NULL,
    `email` varchar(50) NOT NULL,
    `phone` int DEFAULT NULL,
    `address` varchar(255) NOT NULL,
    `date_reg` TIMESTAMP NOT NULL,
    PRIMARY KEY (`id`)
);
CREATE INDEX login ON customers (login(10));
CREATE INDEX email ON customers (email(20));

CREATE TABLE goods (
  `id` int NOT NULL AUTO_INCREMENT,
  `title` varchar(255) NOT NULL,
  `unit_of_measure` varchar(255) NOT NULL,
  `price` int DEFAULT NULL,
  `quantity` int NOT NULL DEFAULT 1,
  `description` varchar(255) NOT NULL,
  PRIMARY KEY (`id`)
);
CREATE INDEX title ON goods (title(20));

CREATE TABLE orders (
  `id` int NOT NULL AUTO_INCREMENT,
  `customers_id` int NOT NULL,
  `price_all` int DEFAULT NULL,
  `payment method` varchar(255) NOT NULL,
  `datetime` TIMESTAMP NOT NULL,
  PRIMARY KEY (`id`)
);
CREATE INDEX customers_id ON orders (customers_id);

CREATE TABLE booking (
    `id` int NOT NULL AUTO_INCREMENT,
    `odrders_id` int NOT NULL,
    `goods_id` int NOT NULL,
    PRIMARY KEY (`id`)
);
CREATE INDEX odrders_id ON booking (odrders_id);
CREATE INDEX goods_id ON booking (goods_id);
